            int t = Convert.ToInt32(Console.ReadLine());

            double Vy0 = V0 * Math.Sin((a * Math.PI) / 180);
            double x = x0 + Vx0 * t;
            double y = y0 + Vy0 * t - (9.81 * Math.Pow(t, 2)) / 2;
            if (y >= 0)
                Console.WriteLine($"x = {x}\ny = {y}");

            {

            }
        }
    }
